#a. by the trapezoidal rule
subintervals = int(input("Enter the number of subintervals you want to divide [1,2]: "))

delta = (2 - 1) / subintervals
x0 = 1
approx_value = 0

for i in range(subintervals + 1):
    x = x0 + delta * i
    if i==0 or i == subintervals:
        approx_value += 1/x
    else:
        approx_value += 2 * (1/x)
        
approx_value *= delta/2
print("Approximated value of the following interval is {}".format(approx_value))
