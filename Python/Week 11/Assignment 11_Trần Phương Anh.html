#exercise 8.14 - cars
def make_car(manufacturer, model_name, **options):
    """Make a dictionary representing a car."""
    car_dict = {
        'manufacturer': manufacturer.title(),
        'model': model_name.title(),
        }
    for option, value in options.items():
        car_dict[option] = value
    return car_dict

car = make_car('suburu', 'outback', color = 'blue', tow_package = True)
print(car)

{'manufacturer': 'Suburu', 'model': 'Outback', 'color': 'blue', 'tow_package': True}

def capitalize_each_word(original_words):
    result = ""
    #tách chuỗi 
    list_of_words = original_words.split()
    for word in list_of_words:
        if len(result) > 0:
            #viết hoa chữ cái đầu và add nó vào list result ban đầu
            #strip() - trả về 1 bản sao của chuỗi ban đầu
            result = result + " " + word.strip().capitalize()
        else:
            result = word.capitalize()
            
    if not result:
        return original_words
    else:
        return result

def insertion_sorts(array):
    for i in range(len(array)):
        for j in range(1, len(array)):
            if array[i] > array[j]:
                temp = array[i]
                array[i] = array[j]
                array[j] = temp
            
            j = j - 1
            i = i - 1
                
array = [23, 12, 2003, 11, 1, 2014]
