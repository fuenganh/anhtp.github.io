class Email:

    def __init__(self, user_name):

            user_name = input('Invalid. Please enter another name: ')
        self.name = user_name
        self.email = user_name + '@dseb.mfe.vn'
        Email.email_created.append(user_name)
        self.inbox = []
    def send(self, userb, mes):
        userb.inbox.append(f'{self.email}: {mes}')
        if len(userb.inbox) > 10:
            userb.inbox.pop(0)
    def number_of_user(self):
        return len(Email.email_created)

class CreditCard:
    create_cardNumber = []
    card_Types = ['Visa', 'MasterCard', 'Amex']
    def __init__(self, cardOwner, cardNumber, cardType, cardBank, cardExpiry):
        self.cardOwner = cardOwner                      #cardOwner: Chủ sở hữu thẻ
        while cardNumber in CreditCard.create_cardNumber:
            cardNumber = input('Invalid. Please enter another number: ')
        self.cardNumber = cardNumber
        CreditCard.create_cardNumber.append(cardNumber) #cardNumber: Số thẻ
        
        while cardType not in CreditCard.card_Types:
            cardType = input('Invalid. Please enter another type: ')
        self.cardType = cardType                        #cardType: loại thẻ visa, mastercard, amex,...                     
        self.cardBank = cardBank                        #cardBank: Ngân hàng phát hành thẻ
        self.cardExpiry = cardExpiry                    #cardExpiry: Ngày hết hạn của thẻ
        self.amount = 0
        self.spent = 0
        self.limit = 2000000000

    def amountMoney(self, amount, spent):
        self.amount = amount
        self.spent = spent
        
    def payment(self, pay = 0):                         #payment: số tiền phải thanh toán
        if pay > self.amount - 500000 or pay > self.limit:
            pay = int(input('Transaction failed. Please enter again: '))
            self.pay = pay
            self.spent += pay
        else:
            pay = int(input('Pay: '))
            self.pay = pay
            self.spent += pay
    
    def __str__(self):
        c = 'Card Information: {} {}\nBank: {}\nOwner of Card: {}\nExpiry Date: {}'.format(self.cardType, self.cardNumber, self.cardBank, self.cardOwner, self.cardExpiry.strftime("%Y-%m-%d"))
        return c
