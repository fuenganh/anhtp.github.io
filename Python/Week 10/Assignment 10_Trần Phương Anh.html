def make_album(artist_name, album_title, tracks = 0):
    """Return a dictionary containing infor about album!"""

#way1: not using recursive
def gcd(m, n):
    if(m==0 or n==0):
        return m+n
    while(m != n):
        if(m > n):
            m = m - n
        else:
            n = n - m
    return m

test = gcd(8, 12)

def check_gcd(a, b):
    if a == 0:
        return b, 0, 1
    else:
        gcd, x, y = check_gcd(b % a, a)
        return gcd, y - (b // a) * x, x
 
 
if __name__ == '__main__':
    gcd, x, y = check_gcd(2, 4)
    print('The GCD is {}'.format(gcd))
    print(f'x = {x}, y = {y}')

def rand(i,n):
    if i == 0:
        s = time.time()
        return int(s)
    else:
        return int((7 * rand(i-1, n) + 1) % n)

def shuffle(alist):
    for i in range(len(alist)-1, 0, -1):
        j = rand(1, i+1)
        alist[i], alist[j] = alist[j], alist[i]
    return alist

list1 = [4, 5, 6, 7, 8, 9, 10]
print(shuffle(list1))
